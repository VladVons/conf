#--- Download image 
#https://www.raspberrypi.org/downloads/raspbian/
wget https://downloads.raspberrypi.org/raspbian_lite_latest -O raspbian-jessie-lite.zip

#--- List usb disks
df -h 
dd bs=4M if=raspbian-jessie.img of=/dev/sdX
sync

#--- Boot Raspberry
user: pi
password: raspberry

#--- Init SSH
service ssh start

#--- configure raspberry
raspi-config
  Menu->Advanced Options->SSH Enable/Disable

#--- Install packages
apt-get install --yes --no-install-recommends mc rsync

#--- Check python version
python -V

#--- install pip
#apt-get install python-pip
wget https://bootstrap.pypa.io/get-pip.py
python get-pip.py

#--- install Flask framework
pip install flask wtforms webhelpers
pip install pyusb
